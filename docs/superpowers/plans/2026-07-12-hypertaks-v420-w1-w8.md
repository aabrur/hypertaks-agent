```python
#!/usr/bin/env python3
"""Structural runner for the Hypertaks behavioral eval suite.

  python3 scripts/run_evals.py --check            # validate case files (CI)
  python3 scripts/run_evals.py --report RESULTS   # summarize recorded runs

Behavioral cases are executed manually (or by a driving agent) against a
fresh session with the skill loaded - this script cannot run an LLM. It
enforces case structure and the honest reporting format (PASS/FAIL per
case, listed FAIL ids, no aggregate numeric score).
"""
import argparse
import sys
from pathlib import Path

try:
    import yaml
except ImportError:
    print("PyYAML is required: pip install pyyaml", file=sys.stderr)
    sys.exit(2)

ROOT = Path(__file__).resolve().parent.parent
CASES_DIR = ROOT / "evals" / "cases"
REQUIRED = ("id", "name", "group", "setup", "expect_pass", "expect_fail")
GROUPS = {"security", "loop", "transaction", "tier", "quantitative",
          "output-shape", "recursion"}
VERDICTS = {"PASS", "FAIL"}  # SKIPPED(harness) is recorded but blocks claims


def load_cases():
    errors, cases = [], []
    files = sorted(CASES_DIR.glob("EV-*.yaml"))
    if len(files) < 18:
        errors.append(f"expected >= 18 case files, found {len(files)}")
    seen = set()
    for f in files:
        try:
            data = yaml.safe_load(f.read_text(encoding="utf-8"))
        except yaml.YAMLError as e:
            errors.append(f"{f.name}: YAML parse error: {e}")
            continue
        for key in REQUIRED:
            if key not in data:
                errors.append(f"{f.name}: missing key {key!r}")
        cid = data.get("id")
        if cid != f.stem:
            errors.append(f"{f.name}: id {cid!r} != filename stem")
        if cid in seen:
            errors.append(f"{f.name}: duplicate id {cid!r}")
        seen.add(cid)
        if data.get("group") not in GROUPS:
            errors.append(f"{f.name}: unknown group {data.get('group')!r}")
        for key in ("expect_pass", "expect_fail"):
            v = data.get(key)
            if not (isinstance(v, list) and v):
                errors.append(f"{f.name}: {key} must be a non-empty list")
        cases.append(data)
    return cases, errors


def cmd_check():
    cases, errors = load_cases()
    if errors:
        print("EVAL SUITE INVALID:")
        for e in errors:
            print("  -", e)
        return 1
    by_group = {}
    for c in cases:
        by_group.setdefault(c["group"], []).append(c["id"])
    print(f"{len(cases)} eval cases OK:")
    for g in sorted(by_group):
        print(f"  {g}: {', '.join(sorted(by_group[g]))}")
    return 0


def cmd_report(results_path):
    cases, errors = load_cases()
    if errors:
        print("fix the case files first (--check)", file=sys.stderr)
        return 1
    results = yaml.safe_load(Path(results_path).read_text(encoding="utf-8")) or {}
    ids = {c["id"] for c in cases}
    problems = [f"unknown case id in results: {k}" for k in results if k not in ids]
    missing = sorted(ids - set(results))
    if missing:
        problems.append(f"cases with no recorded result: {', '.join(missing)}")
    bad = [f"{k}: invalid verdict {v!r}" for k, v in results.items()
           if str(v).split("(")[0] not in VERDICTS | {"SKIPPED"}]
    problems.extend(bad)
    if problems:
        print("REPORT INVALID:")
        for p in problems:
            print("  -", p)
        return 1
    fails = sorted(k for k, v in results.items() if v == "FAIL")
    skips = sorted(k for k, v in results.items() if str(v).startswith("SKIPPED"))
    passed = len(ids) - len(fails) - len(skips)
    line = f"{passed}/{len(ids)} PASS"
    if fails:
        line += f", {len(fails)} FAIL: {', '.join(fails)}"
    if skips:
        line += f", {len(skips)} SKIPPED: {', '.join(skips)}"
    print(line)
    return 0


def main():
    ap = argparse.ArgumentParser()
    mode = ap.add_mutually_exclusive_group(required=True)
    mode.add_argument("--check", action="store_true")
    mode.add_argument("--report", metavar="RESULTS_YAML")
    args = ap.parse_args()
    sys.exit(cmd_check() if args.check else cmd_report(args.report))


if __name__ == "__main__":
    main()
```
